221  void pauseState(ExecutionState &state){
222  assert(std::find(pausedStates.begin(), pausedStates.end(), &state) == pausedStates.end());
223  pausedStates.push_back(&state);
224  baseSearcher->removeState(&state);
225  }
226 
228  void continueState(ExecutionState &state){
229  auto it = std::find(pausedStates.begin(), pausedStates.end(), &state);
230  assert( it != pausedStates.end());
231  pausedStates.erase(it);
232  baseSearcher->addState(&state);
233  }
234 
235  ExecutionState &selectState();

237  void update(ExecutionState *current,
238  const std::vector<ExecutionState *> &addedStates,
239  const std::vector<ExecutionState *> &removedStates) {
240  // We have to check if the current execution state was just deleted, as to
241  // not confuse the nurs searchers
242  if (std::find(pausedStates.begin(), pausedStates.end(), current) ==
243  pausedStates.end()) {
244  baseSearcher->update(current, addedStates, removedStates);
245  }
246  }
247 
248  bool empty() { return baseSearcher->empty(); }
249  void printName(llvm::raw_ostream &os) {
250  os << "MergingSearcher\n";
251  }
252  };
